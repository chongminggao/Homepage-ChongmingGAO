\cventry{ICDM '19}{\small Generating Reliable Friends via Adversarial Training to Improve Social Recommendation}{}{}{}{\small \cu{余俊良, 高旻, 阴红志, 李俊东, \underline{高崇铭}, 王覃泳}\vspace{1mm}\\

}
